HTML Template & Example Code

Below is a short snippet you can reuse inside your webpage for showing clusters or results. Copy-paste into your project.

<!-- Simple CSV download from in-memory data -->
<script>
const csvContent = `Area,Murder,Theft,Assault,Fraud\nA1,10,50,25,5\nA2,2,15,3,1\nA3,6,30,10,2\nA4,20,70,40,6`;
function downloadCSV(){
  const blob = new Blob([csvContent],{type:'text/csv'});
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = 'crime_sample.csv';
  document.body.appendChild(a); a.click(); a.remove(); URL.revokeObjectURL(url);
}
document.getElementById('downloadCsv').addEventListener('click', downloadCSV);
</script>
